def muzik_yukle(dosya_adi='music.txt'):
    """music.txt dosyasƒ±ndan m√ºzik dizisini y√ºkle"""
    try:
        script_dir = os.path.dirname(os.path.abspath(__file__))
        dosya_yolu = os.path.join(script_dir, dosya_adi)
        
        if not os.path.exists(dosya_yolu):
            print(f"‚ùå Hata: {dosya_adi} dosyasƒ± bulunamadƒ±!")
            print(f"üìÅ Aranan konum: {dosya_yolu}")
            print(f"üí° √á√∂z√ºm: music.txt dosyasƒ±nƒ± ≈üu konuma kaydedin: {script_dir}")
            
            try:
                dosyalar = os.listdir(script_dir)
                print(f"\nüìÇ Mevcut dizindeki dosyalar:")
                for d in dosyalar:
                    print(f"  - {d}")
            except:
                pass
            
            return []
        
        with open(dosya_yolu, 'r', encoding='utf-8') as f:
            icerik = f.read()
        
        # "music = [" ile ba≈ülƒ±yorsa, sadece liste kƒ±smƒ±nƒ± al
        if 'music = [' in icerik:
            baslangic = icerik.find('[')
            bitis = icerik.rfind(']') + 1
            liste_str = icerik[baslangic:bitis]
        else:
            liste_str = icerik
        
        # String'i Python listesine √ßevir
        music = ast.literal_eval(liste_str)
        
        # Format kontrol√º - yeni format (zaman, nota, enstruman)
        if music and len(music[0]) == 3 and isinstance(music[0][0], (int, float)):
            print(f"‚úÖ {len(music)} nota y√ºklendi (mutlak zamanlama)")
        else:
            print(f"‚ùå Hata: Ge√ßersiz m√ºzik formatƒ±!")
            print(f"üí° Format: (zaman_saniye, nota, enstruman)")
            return []
        
        return music
    
    except FileNotFoundError:
        print(f"‚ùå Hata: {dosya_adi} dosyasƒ± bulunamadƒ±!")
        return []
    except Exception as e:
        print(f"‚ùå M√ºzik dosyasƒ± okunurken hata: {e}")
        return []


def nota_cal(note_name, instrument, mcr):
    """Belirtilen enstr√ºman ve notayƒ± √ßal"""
    y_coord = INSTRUMENT_Y.get(instrument)
    z_coord = NOTA_Z_KOORDS.get(instrument, {}).get(note_name)
    
    if y_coord is None or z_coord is None:
        return
    
    try:
        with rcon_lock:
            cmd = f'setblock 0 {y_coord} {z_coord} {blok}'
            mcr.command(cmd)
            time.sleep(0.05)
            cmd = f'setblock 0 {y_coord} {z_coord} air'
            mcr.command(cmd)
    except Exception as e:
        print(f"Hata ({note_name}/{instrument}): {e}")


def muzik_cal(melody, mcr):
    """M√ºziƒüi polifonik olarak √ßalar - mutlak zamanlama ile"""
    if not melody:
        return
    
    # Zamanƒ± aynƒ± olan notalarƒ± grupla (20ms tolerans)
    nota_gruplari = defaultdict(list)
    for zaman, note, instrument in melody:
        zaman_anahtari = round(zaman * 50) / 50  # 20ms hassasiyet
        nota_gruplari[zaman_anahtari].append((note, instrument))
    
    # Sƒ±ralƒ± zamanlarƒ± al
    zamanlar = sorted(nota_gruplari.keys())
    
    baslangic = time.time()
    
    for hedef_zaman in zamanlar:
        # Hedef zamana kadar bekle
        simdiki_zaman = time.time() - baslangic
        bekleme = hedef_zaman - simdiki_zaman
        
        if bekleme > 0:
            time.sleep(bekleme)
        
        # Bu zamandaki t√ºm notalarƒ± paralel √ßal
        notalar = nota_gruplari[hedef_zaman]
        
        if len(notalar) > 1:
            # Birden fazla nota varsa thread'lerle paralel √ßal
            threads = []
            for note, instrument in notalar:
                t = threading.Thread(target=nota_cal, args=(note, instrument, mcr))
                t.start()
                threads.append(t)
            
            # T√ºm notalarƒ±n √ßalmasƒ±nƒ± bekle
            for t in threads:
                t.join()
        else:
            # Tek nota
            note, instrument = notalar[0]
            nota_cal(note, instrument, mcr)
